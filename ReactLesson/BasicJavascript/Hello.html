二，JavaScript变量声明

                开始实验吧！
                
                    var a = 1; // 全局（函数内部？）作用域
                    let b = 2;  // 块级作用域
                    const c = 3; // 常量
                
                实验一，在函数外调用var变量
                
                    var a = 1;
                    function test(){
                        console.log("实验一:"+a);
                    }
                    test();
                
                结果是输出1

                实验二，在函数内调用var变量
                
                    var a = 1;
                    function test(){
                        var a = 2;
                        console.log(a);
                    }
                    test();
                    console.log(a);
                
                结果是输出2，后面是1

三，JS模板字符串

                
                    let a=first;
                    console.log(`${a},Welcome to React!`);
                
                类似于bash模板字符串，但是需要使用反引号
                可以免去使用字符串拼接的麻烦
                还可以像pre元素一样，存储预格式化的文本
                
                    let a=first;
                    console.log(`
                        ${a}
                        Welcome to React!
                    `);
                
                在处理html的时候可以实现格式化！（或者文本等等）

四，js函数

                函数声明
                
                    test();
                    //可以在函数声明前调用函数，函数声明类似于var，有作用域提升
                    function test(){
                        console.log("函数声明");
                    }
                
                函数表达式
                
                    let test = function(){
                        console.log("函数表达式");
                    }
                    test();//但是这个不能提前调用
                
                箭头函数
                
                    let test = () => {
                        console.log("箭头函数"); // 箭头函数没有this，所以this的作用域还是父函数（如果有的话）！箭头指向的就是返回值
                    }
                    test();
                
                函数参数
                
                    let test = (a,b) => {
                        console.log(a+b);
                    }
                    test(1,2);
                
                函数返回值
                
                    let test = (a,b) => {
                        return a+b;
                    }
                    console.log(test(1,2));
                
                函数返回对象
                
                    let test = (a,b) => {
                        return {a,b};
                    }
                    console.log(test(1,2));
                    //或者
                    let test = (a,b) => (
                        {
                            a,
                            b
                        });
                
                函数参数默认值
                
                    let test = (a,b=2) => {
                        return a+b;
                    }
                    console.log(test(1));
                
                函数参数解构
                
                    let test = ({a,b}) => {
                        return a+b;
                    }
                    console.log(test({a:1,b:2,c:3})); // 只取a和b
                
                函数参数解构，数组
                
                    let test = ([a,b]) => {
                        return a+b;
                    }
                    console.log(test([1,2])); // 只取第一个和第二个
                    let test2 = ([a,b,...rest]) => {
                        return a+b+rest.join(","); // rest是数组
                    }
                    console.log(test2([1,2,3,4,5]));
                
                展开运算符
                
                    let arr1 = [1,2,3];
                    let arr2 = [4,5,6];
                    let arr3 = [...arr1,...arr2];
                    console.log(arr3);
                    //处理对象
                    let obj1 = {name:'zhangsan',age:18};
                    let obj2 = {sex:'male',height:180};
                    let obj3 = {...obj1,...obj2};
                    console.log(obj3);
五，js异步编程 
                1,使用fetch获得promise

                    }).catch((err)=>{

                    });

                    async function getUserInfo(){
